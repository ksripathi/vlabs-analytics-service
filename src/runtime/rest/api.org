@api.route('/<analytics>',methods= ['GET','POST'])
def add_openedx_usage_analytics(analytics):
    data = request.data
    fetch_data = analytics
    data_list = fetch_data.split(",")
    data_dict = {}
    today = str(datetime.datetime.today()).split()[0]
    data_dict["student_id"] = data_list[0]
    data_dict["lab_id"] = data_list[1].lower()
    LDS_API = "%s/labs?lab_id=%s" % (LDS_URL, data_dict['lab_id'])
    r = requests.get(LDS_API)
    if r.status_code == 200:
        data_dict["lab_name"] = r.json()['lab_name']
    else:
        data_dict["lab_name"] = data_list[2]
    data_dict["experiment_id"] = data_list[3].lower()
    LDS_API = "%s/experiments?exp_id=%s" % (LDS_URL, data_dict['experiment_id'])
    r = requests.get(LDS_API)
    if r.status_code == 200:
        data_dict["experiment_name"] = r.json()['exp_name']
    else:
        data_dict["experiment_name"] = data_list[4]
    data_dict["date_of_experiment"] = today
    data_dict["time_of_experiment"] = time.strftime("%H:%M")
    data_dict["ip_address"] = request.environ.get('HTTP_X_REAL_IP',\
                                                  request.remote_addr)

    res = requests.get("http://freegeoip.net/json/" + \
                       data_dict["ip_address"])
    data = res.text
    ip_dir = ast.literal_eval(data)
    data_dict["region"] = ip_dir["region_name"]

    json_data = json.dumps(data_dict)
    headers = {'Content-Type': 'application/json'}
    index = "vlabs"
    type = "openedx_usage"
    ANALYTICS_API = "%s/%s/%s" % (ANALYTICS_URL, index, type)
    res = requests.post(ANALYTICS_API, data=json_data, headers=headers)
    if res.status_code == 200 or res.status_code == 201:
        return "success"
    else:
        return "failed"

def get_ip_info(ip_address, res):
    data = {}
    clients_info = res.json()['hits']['hits']

    for client_info in clients_info:
        current_app.logger.debug("ip_address after call= %s db_ip_info =%s" % (client_info['_source']['clientip'], ip_address))

            data['clientip'] = ip_address

                    }
                }
            }
        }

        index = "vlabs"
        type = "test_api1_user_analytics_%s" % (current_date)
        ANALYTICS_API = "%s/%s/%s/_search?size=10000" % (ANALYTICS_URL, index, type)
        current_app.logger.debug("analytics_api =%s" % (ANALYTICS_API))
        active_users = {}
        total_active_users = 0
        try:
            res = requests.post(ANALYTICS_API, data=json.dumps(data), headers=headers)
            if res.status_code == 200:

                clients_ip_info = res.json()['aggregations']['client_ips']['buckets']
                details = []
                current_app.logger.debug("clients_ip_info =%s" % (clients_ip_info))
                for client_action in clients_ip_info:

                    if len(client_action['client_status']['buckets']) == 1:
                        current_app.logger.debug("buckets with size 1 =%s" % (len(client_action['client_status']['buckets'])))
                        current_app.logger.debug("ip_address before call =%s" % (client_action['key']))
                        ip_details = get_ip_info(client_action['key'], res)
                        current_app.logger.debug("ip details =%s" % (ip_details))
                        active_users_count = client_action['doc_count']
                        ip_details['count'] = int(active_users_count)
                        total_active_users = total_active_users + ip_details['count']

                    elif len(client_action['client_status']['buckets']) == 2:
                        current_app.logger.debug("buckets with size 2 =%s" % (len(client_action['client_status']['buckets'])))
                        ip_details = get_ip_info(client_action['key'], res)
                        login_count = int(client_action['client_status']['buckets'][0]['doc_count'])
                        logout_count = int(client_action['client_status']['buckets'][1]['doc_count'])
                        active_users_count = login_count - logout_count
                        if active_users_count > 0:
                            ip_details['count'] = int(active_users_count)
                            total_active_users = total_active_users + ip_details['count']
                        if active_users_count == 0:
                            ip_details['count'] = 0


                    else:
                        pass
                    current_app.logger.debug("details before append %s" % (ip_details))
                    details.append(ip_details)
                current_app.logger.debug("for loop end details =%s" % (details))
                active_users['current_logged_in_users'] = {}
                active_users['date'] = current_date
                active_users['current_logged_in_users']['details'] = details
                active_users['current_logged_in_users']['total'] = total_active_users
                current_app.logger.debug("active_users =%s" % (active_users))
                return jsonify(active_users)

            else:
                abort(res.status_code)
        except Exception as e:
            abort(str(e))
